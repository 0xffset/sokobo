\begin{DoxyItemize}
\item {\bfseries{Symbolic \doxylink{class_expression}{Expression} Manipulation}}:\+ Variables, constants, binary operations, functions
\item {\bfseries{\doxylink{class_polynomial}{Polynomial} Arithmetic}}:\+ Addition, multiplication, division, GCD, factorization
\item {\bfseries{Complex Number Operations}}:\+ All basic operations plus advanced functions
\item {\bfseries{\doxylink{class_matrix}{Matrix} Operations}}:\+ Determinant, inverse, eigenvalues, decompositions
\item {\bfseries{\doxylink{class_calculus}{Calculus}}}:\+ Symbolic differentiation, numerical integration, series expansions
\item {\bfseries{Transforms}}:\+ Laplace transforms, Fourier transforms (DFT/\+\+FFT)
\item {\bfseries{Differential Equations}}:\+ ODE and PDE solvers
\item {\bfseries{Numerical Methods}}:\+ Root finding, optimization, interpolation
\end{DoxyItemize}\hypertarget{md__r_e_a_d_m_e_autotoc_md3}{}\doxysubsection{\texorpdfstring{Advanced Features}{Advanced Features}}\label{md__r_e_a_d_m_e_autotoc_md3}

\begin{DoxyItemize}
\item Taylor and Chebyshev series expansions
\item Multiple integration methods (Trapezoidal, Simpson\textquotesingle{}s, Gaussian quadrature, Monte Carlo)
\item Various ODE solvers (Euler, Runge-\/\+Kutta, Adams-\/\+Bashforth)
\item PDE solvers for heat, wave, and Laplace equations
\item Linear system solvers (Gaussian elimination, LU decomposition, iterative methods)
\item Statistical functions and regression analysis
\item Special functions (Gamma, Beta, Error, Bessel, Legendre, Hermite)
